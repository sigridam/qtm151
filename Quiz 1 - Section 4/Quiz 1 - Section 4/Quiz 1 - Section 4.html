list_csv = ["data_quiz\wdi_co2_2005.csv", "data_quiz\wdi_co2_2010.csv", "data_quiz\wdi_co2_2015.csv"]
index = 0
t=len(list_csv)
for i in range(t):

    co2_gasfuel_kt = dataset["co2_gasfuel_kt"]
    co2_liquidfuel_kt = dataset["co2_liquidfuel_kt"]

    plt.xlabel("total co2")
    plt.ylabel("frequency")
    plt.show()
